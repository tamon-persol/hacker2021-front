# Grid

## Usage
```jsx
import { Grid } from 'components/atoms/Grid'
import { Box } from 'components/atoms/Box';

<Grid container spacing={2}>
  <Grid item xs={6}>
    <Box
      bgcolor="#f0f8ff"
      color="#808080"
      padding={4}
      textAlign="center"
      borderRadius={6}
    >
      xs6
    </Box>
  </Grid>
  <Grid item xs={6}>
    <Box
      bgcolor="#f0f8ff"
      color="#808080"

</Grid>
```
